\documentclass{article}
\usepackage{hyperref}
\usepackage{fullpage}
\usepackage{alltt}

\newenvironment{MYverbatim}{\begin{flushright}%
    \begin{minipage}[h]{0.9\linewidth}%
      \begin{alltt}}%
      {\end{alltt}\end{minipage}\end{flushright}}
\renewenvironment{verbatim}{\begin{MYverbatim}}{%
    \end{MYverbatim}\ignorespacesafterend% as suggested above
}

\setlength{\parindent}{0pt}
\setlength{\parskip}{1ex}

\begin{document}
\input{|"pandoc -f markdown -t latex script.md"}
\end{document}
